-- ============================================================
-- OPPSLAGSTABELLER
-- ============================================================

CREATE TABLE Kjonn (
    Id      SERIAL PRIMARY KEY,
    Navn    TEXT NOT NULL,
    Kortform TEXT NOT NULL
);

CREATE TABLE Klubb (
    Id              SERIAL PRIMARY KEY,
    KlubbNavn       TEXT NOT NULL,
    KlubbKortNavn   TEXT NOT NULL,
    ErAktiv         BOOLEAN NOT NULL DEFAULT TRUE,
    LogoUrl         TEXT
);

CREATE TABLE Klasse (
    Id       SERIAL PRIMARY KEY,
    Navn     TEXT NOT NULL,
    Kortform TEXT NOT NULL,
    ErAktiv  BOOLEAN NOT NULL DEFAULT TRUE
);

CREATE TABLE Gruppe (
    Id   SERIAL PRIMARY KEY,
    Navn TEXT NOT NULL  -- "A", "B"
);

CREATE TABLE StevneType (
    Id       SERIAL PRIMARY KEY,
    Navn     TEXT NOT NULL,  -- "NM", "NC", "SNC", "DNC", "Lokalt"
    Kortform TEXT NOT NULL
);

CREATE TABLE Kastemetode (
    Id               SERIAL PRIMARY KEY,
    Navn             TEXT NOT NULL,
    Beskrivelse      TEXT,
    ErAktiv          BOOLEAN NOT NULL DEFAULT TRUE,
    ErNorgesranking  BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE Kategori (
    Id            SERIAL PRIMARY KEY,
    Navn          TEXT NOT NULL,  -- "Singel", "Par", "Mix", "Lag", "X-kast", "Kongelag", "Hesteskogolf"
    ErLagbasert   BOOLEAN NOT NULL DEFAULT FALSE,
    BrukerRinger  BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE NorgescupPoeng (
    Id              SERIAL PRIMARY KEY,
    Plassering      INT NOT NULL,
    Poeng           INT NOT NULL,
    GjelderFraAar   INT NOT NULL,
    GjelderTilAar   INT  -- NULL = gjelder fortsatt
);

-- ============================================================
-- KJERNETABELLER
-- ============================================================

CREATE TABLE Arrangement (
    Id                      SERIAL PRIMARY KEY,
    Navn                    TEXT NOT NULL,
    Sted                    TEXT,
    StartDato               DATE NOT NULL,
    SluttDato               DATE,
    ArrangorKlubbId         INT REFERENCES Klubb(Id),
    InnbydelseUrl           TEXT,
    InnbydelseFilnavn       TEXT,
    InnbydelseOppdatert     TIMESTAMPTZ
);

-- Kaster må opprettes før Stevne pga. KontaktKasterId FK
CREATE TABLE Kaster (
    Id            SERIAL PRIMARY KEY,
    Medlemsnummer INT,
    Fornavn       TEXT NOT NULL,
    Etternavn     TEXT NOT NULL,
    Telefon       TEXT,
    Epost         TEXT,
    KjonnId       INT NOT NULL REFERENCES Kjonn(Id),
    KlubbId       INT REFERENCES Klubb(Id),
    KlasseId      INT REFERENCES Klasse(Id),
    ErAktiv       BOOLEAN NOT NULL DEFAULT TRUE,
    BildeUrl      TEXT
);

CREATE TABLE Stevne (
    Id                          SERIAL PRIMARY KEY,
    ArrangementId               INT NOT NULL REFERENCES Arrangement(Id),
    KategoriId                  INT NOT NULL REFERENCES Kategori(Id),
    StevneTypeId                INT REFERENCES StevneType(Id),
    InnledendeKastemetodeId     INT REFERENCES Kastemetode(Id),
    AvsluttendeKastemetodeId    INT REFERENCES Kastemetode(Id),
    KontaktKasterId             INT REFERENCES Kaster(Id),
    Juryleder                   TEXT,
    ErNM                        BOOLEAN NOT NULL DEFAULT FALSE,
    ErNorgesranking             BOOLEAN NOT NULL DEFAULT FALSE,
    HarKlasseInndeling          BOOLEAN NOT NULL DEFAULT FALSE,
    ErFullfort                  BOOLEAN NOT NULL DEFAULT FALSE,
    ErEkskludertFraRekorder     BOOLEAN NOT NULL DEFAULT FALSE,
    ResultatUrl                 TEXT,
    ResultatFilnavn             TEXT,
    ResultatOppdatert           TIMESTAMPTZ
);

CREATE TABLE Resultat (
    Id              SERIAL PRIMARY KEY,
    StevneId        INT NOT NULL REFERENCES Stevne(Id),
    KasterId        INT REFERENCES Kaster(Id),
    KlubbId         INT REFERENCES Klubb(Id),    -- Historisk klubb
    KlasseId        INT REFERENCES Klasse(Id),   -- Historisk klasse
    GruppeId        INT REFERENCES Gruppe(Id),
    Plassering      INT,
    ErPremie        BOOLEAN,
    NorgescupPoeng  REAL
);

-- Én-til-én: kun for Gloppen / Nordhordlandsmetoden
CREATE TABLE ResultatStandard (
    ResultatId      INT PRIMARY KEY REFERENCES Resultat(Id) ON DELETE CASCADE,
    KampPoeng       REAL,   -- 2, 1.5, 1 eller 0
    SkarInnledende  INT
);

-- Én-til-én: kun for X-kast, Kongelag, Hesteskogolf
CREATE TABLE ResultatSpesial (
    ResultatId              INT PRIMARY KEY REFERENCES Resultat(Id) ON DELETE CASCADE,
    PoengMinimatch          INT,
    PoengXHalvmatch         INT,
    PoengXHeilmatch         INT,
    PoengKongelag           INT,
    PoengGolf               INT,
    AntallRingMinimatch     INT,
    AntallRingHalvmatch     INT,
    AntallRingHeilmatch     INT,
    AntallRingKongelag      INT
);

-- ============================================================
-- KAMP-TABELLER (valgfritt – brukes av scoreappen)
-- ============================================================

CREATE TABLE Kamp (
    Id              SERIAL PRIMARY KEY,
    StevneId        INT NOT NULL REFERENCES Stevne(Id),
    KastemetodeId   INT NOT NULL REFERENCES Kastemetode(Id),
    ErInnledende    BOOLEAN NOT NULL DEFAULT FALSE,
    ErAvsluttende   BOOLEAN NOT NULL DEFAULT FALSE,
    RundeNummer     INT
);

CREATE TABLE KampSide (
    Id          SERIAL PRIMARY KEY,
    KampId      INT NOT NULL REFERENCES Kamp(Id) ON DELETE CASCADE,
    SideNummer  INT NOT NULL CHECK (SideNummer IN (1, 2)),
    TotalPoeng  INT,
    TotalRinger INT,
    ErVinner    BOOLEAN
);

CREATE TABLE KampSideDeltaker (
    Id              SERIAL PRIMARY KEY,
    KampSideId      INT NOT NULL REFERENCES KampSide(Id) ON DELETE CASCADE,
    KasterId        INT NOT NULL REFERENCES Kaster(Id),
    ErStartspiller  BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE KampOmgang (
    Id            SERIAL PRIMARY KEY,
    KampSideId    INT NOT NULL REFERENCES KampSide(Id) ON DELETE CASCADE,
    OmgangNummer  INT NOT NULL,
    Poeng         INT NOT NULL,
    AntallRinger  INT
);

-- ============================================================
-- INDEKSER
-- ============================================================

-- Resultat – hyppig filtrering per stevne
CREATE INDEX idx_resultat_stevne    ON Resultat(StevneId);
CREATE INDEX idx_resultat_kaster    ON Resultat(KasterId);
CREATE INDEX idx_resultat_plassering ON Resultat(StevneId, Plassering);

-- Stevne – filtrering per arrangement
CREATE INDEX idx_stevne_arrangement ON Stevne(ArrangementId);

-- Arrangement – sortering på dato
CREATE INDEX idx_arrangement_dato   ON Arrangement(StartDato DESC);

-- Kamp – oppslag per stevne
CREATE INDEX idx_kamp_stevne        ON Kamp(StevneId);

-- KampOmgang – oppslag per side
CREATE INDEX idx_kampomgang_side    ON KampOmgang(KampSideId);

-- NorgescupPoeng – årsbasert oppslag
CREATE INDEX idx_norgescup_aar      ON NorgescupPoeng(GjelderFraAar, GjelderTilAar);

-- ============================================================
-- VIEW: Flattened ResultatFull (praktisk for frontend/API)
-- ============================================================

CREATE VIEW ResultatFull AS
SELECT
    r.Id,
    r.StevneId,
    r.KasterId,
    r.KlubbId,
    r.KlasseId,
    r.GruppeId,
    r.Plassering,
    r.ErPremie,
    r.NorgescupPoeng,
    -- Standard-felt (Gloppen / Nordhordland)
    rs.KampPoeng,
    rs.SkarInnledende,
    -- Spesial-felt (X-kast, Kongelag, Hesteskogolf)
    rsp.PoengMinimatch,
    rsp.PoengXHalvmatch,
    rsp.PoengXHeilmatch,
    rsp.PoengKongelag,
    rsp.PoengGolf,
    rsp.AntallRingMinimatch,
    rsp.AntallRingHalvmatch,
    rsp.AntallRingHeilmatch,
    rsp.AntallRingKongelag
FROM Resultat r
LEFT JOIN ResultatStandard  rs  ON rs.ResultatId  = r.Id
LEFT JOIN ResultatSpesial   rsp ON rsp.ResultatId = r.Id;
